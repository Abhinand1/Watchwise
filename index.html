        function createChipHTML(value, type, isSelected) {

            const selectedClass = isSelected ? 'neon-chip-selected' : 'neon-chip-default text-slate-300';

            return `

        }

        function renderChips(containerId, list, selectedList, type) {

            const container = document.getElementById(containerId);

            if (!container) return;

            container.innerHTML = list.map(item => {

                const isSelected = selectedList.includes(item);

                return createChipHTML(item, type, isSelected);

            }).join('');

            checkRequirements();

        }

        function toggleSelection(button) {

            const value = button.getAttribute('data-value');

            const type = button.getAttribute('data-type');

           

            let list = type === 'category' ? selectedCategories : selectedLanguages;

            const index = list.indexOf(value);

            if (index === -1) {

                list.push(value);

                button.classList.remove('neon-chip-default', 'text-slate-300');

                button.classList.add('neon-chip-selected');

            } else {

                list.splice(index, 1);

                button.classList.remove('neon-chip-selected');

                button.classList.add('neon-chip-default', 'text-slate-300');

            }

            if (type === 'category') selectedCategories = list;

            if (type === 'language') selectedLanguages = list;

           

            checkRequirements();

        }

        // --- Gemini API & Flow Control ---

       

        async function fetchRecommendations(query) {

            const systemPrompt = `You are a world-class movie and series recommendation engine called WatchWise. Based on the user's query, find and list exactly ${RESULTS_PER_PAGE}, highly-rated, relevant titles. For each title, you MUST provide: 1. The Title. 2. A concise 2-line Synopsis. 3. The Release Date (only the year is fine). 4. The IMDb Score as a single number (e.g., 8.8). 5. The Rotten Tomatoes Score as a percentage (e.g., 87%). Present the results clearly, with each movie/series output separated by two new lines. Do not add any conversational text, headers, or numbering. Label each field explicitly for easy parsing (e.g., "Title: Inception", "IMDb Score: 8.8", "RT Score: 87%"). IF a score or date is not available, use "N/A".`;

           

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = {

                contents: [{ parts: [{ text: query }] }],

                tools: [{ "google_search": {} }],

                systemInstruction: { parts: [{ text: systemPrompt }] },

            };

            const maxRetries = 3;

            for (let attempt = 0; attempt < maxRetries; attempt++) {

                try {

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    if (response.status === 429 && attempt < maxRetries - 1) {

                        const delay = Math.pow(2, attempt) * 1000 + Math.random() * 1000;

                        console.warn(`Rate limit hit. Retrying in ${delay / 1000}s...`);

                        await new Promise(resolve => setTimeout(resolve, delay));

                        continue;

                    }

                    if (!response.ok) {

                        throw new Error(`API call failed with status: ${response.status}`);

                    }

                   

                    const result = await response.json();

                    const text = result.candidates?.[0]?.content?.parts?.[0]?.text || "";

                   

                    if (!text.trim()) {

                        throw new Error("Gemini returned an empty response.");

                    }

                   

                    return text;

                } catch (error) {

                    console.error("Fetch attempt failed:", error);

                    if (attempt === maxRetries - 1) {

                        throw new Error("Failed to get recommendations after multiple retries.");

                    }

                }

            }

        }

       

        function parseRecommendations(responseText) {

            const blocks = responseText.trim().split(/\n\n+/);

            const recommendations = [];

            blocks.forEach(block => {

                const lines = block.trim().split('\n');

                let rec = { title: '', synopsis: '', releaseDate: 'N/A', imdbScore: 'N/A', rtScore: 'N/A'};

                lines.forEach(line => {

                    const firstColonIndex = line.indexOf(':');

                    if (firstColonIndex === -1) return;

                    const key = line.substring(0, firstColonIndex).trim().toLowerCase();

                    const value = line.substring(firstColonIndex + 1).trim();

                    if (key === 'title') {

                        rec.title = value;

                    } else if (key === 'synopsis') {

                        rec.synopsis = value;

                    } else if (key.includes('release date')) {

                        rec.releaseDate = value.match(/\d{4}/)?.[0] || value || 'N/A';

                    } else if (key.includes('imdb score')) {

                        rec.imdbScore = value.match(/(\d+(\.\d)?)/)?.[0] || 'N/A';

                    } else if (key.includes('rotten tomatoes score') || key.includes('rt score')) {

                        rec.rtScore = value.match(/(\d+%)|(\d+)/)?.[0] || 'N/A';

                    }

                });

                if (rec.title && rec.title !== 'Title' && rec.title.length > 3) {

                    recommendations.push(rec);

                }

            });

            // Filter duplicates (based on title)

            const existingTitles = Array.from(recList.querySelectorAll('h3')).map(h => h.textContent.trim());

            return recommendations.filter(rec => !existingTitles.includes(rec.title));

        }

        async function handleRecommendationRequest(isMoreRequest, isRandomRequest = false) {

            if (!generateButton || !randomButton) return console.error("DOM elements not ready for request handling.");

            if (!isRandomRequest && selectedCategories.length === 0 && !isMoreRequest) {

                return showMessage("Please select at least one Category before generating recommendations.");

            }

            loadingBox.classList.remove('hidden');

            generateButton.disabled = true;

            randomButton.disabled = true;

            if(loadMoreButton) {

                loadMoreButton.disabled = true;

                if(isMoreRequest) {

                    loadMoreButton.innerHTML = `<i data-lucide="loader-2" class="w-4 h-4 mr-1 inline-block animate-spin"></i> Loading More`;

                }

            }

           

            // Hide the load more button while fetching

            if(loadMoreContainer) loadMoreContainer.classList.add('hidden');

            let query;

            if (isRandomRequest) {

                query = `Find exactly ${RESULTS_PER_PAGE} highly-rated, completely random ${selectedType}s across all genres and periods.`;

            } else {

                const categoryList = selectedCategories.join(', ');

                query = `Find exactly ${RESULTS_PER_PAGE} highly-rated ${selectedType}s in the ${categoryList} genre.`;

               

                if (selectedLanguages.length > 0) {

                    const languageList = selectedLanguages.join(' or ');

                    query += ` CRITICAL CONSTRAINT: The results MUST ONLY contain ${selectedType}s available in the ${languageList} language. DO NOT include any titles not in the specified languages.`;

                }

                if (selectedPeriod !== PERIOD_MAP[0]) {

                    query += ` with a release date or period of ${selectedPeriod}.`;

                }

            }

           

            if (isMoreRequest) {

                // Remove the Load More button placeholder before appending new results

                const moreButtonEl = document.getElementById('load-more-container');

                if (moreButtonEl) moreButtonEl.remove();

                const existingTitles = Array.from(recList.querySelectorAll('h3')).map(h => h.textContent.trim());

                query += ` Provide ${RESULTS_PER_PAGE} *new* results that are different from the ones already generated. Titles already generated: ${existingTitles.slice(-5).join(', ')}.`;

            } else {

                // If it's a new search, clear the list

                 recList.innerHTML = '';

            }

            try {

                const rawResults = await fetchRecommendations(query);

                const newRecommendations = parseRecommendations(rawResults);

               

                let success = false;

                if (newRecommendations.length > 0) {

                    newRecommendations.forEach(rec => {

                        recList.insertAdjacentHTML('beforeend', createCardHTML(rec));

                        recommendationCount++;

                    });

                    success = true;

                   

                    if (!isMoreRequest) {

                        inputScreen.classList.add('hidden');

                        resultsScreen.classList.remove('hidden');

                       

                        // Hide the tagline when moving to results screen

                        if (taglineQuoteContainer) taglineQuoteContainer.classList.add('hidden');

                       

                        window.scrollTo(0, 0);

                        updateFilterSummary();

                        showMessage(`Found ${recommendationCount} initial recommendations!`);

                    } else {

                        showMessage(`Added ${newRecommendations.length} more recommendations!`);

                    }

                } else if (!isMoreRequest) {

                    recList.innerHTML = '<p class="text-center text-slate-400 p-6 rounded-xl bg-gray-800">We couldn\'t find suitable results. Try different categories or "Fully Random"!</p>';

                    showMessage("No results found. Try adjusting your filters!");

                } else {

                    showMessage("No further unique recommendations found right now.");

                }

               

               

                // Re-insert the load more button at the bottom of the list if results were found.

                if (success && recList && loadMoreContainer) {

                    recList.appendChild(loadMoreContainer);

                    loadMoreContainer.classList.remove('hidden');

                }

               

                lucide.createIcons();

            } catch (error) {

                console.error("WatchWise Error:", error);

               

                if (!isMoreRequest && recList.innerHTML.trim() === '') {

                     recList.innerHTML = '<p class="text-center text-slate-400 p-6 rounded-xl bg-gray-800">Connection Error. Failed to fetch recommendations. Try again in a moment!</p>';

                     showMessage(`Connection Error: Failed to fetch live recommendations.`);

                     inputScreen.classList.add('hidden');

                     resultsScreen.classList.remove('hidden');

                } else if (isMoreRequest) {

                     showMessage(`Connection Error: Failed to load more recommendations.`);

                }

            } finally {

                // End Loading state

                loadingBox.classList.add('hidden');

                generateButton.disabled = selectedCategories.length === 0 && !isRandomRequest;

                randomButton.disabled = false;

                if(loadMoreButton) {

                     loadMoreButton.disabled = false;

                     loadMoreButton.innerHTML = `<i data-lucide="plus" class="w-4 h-4 mr-1 inline-block"></i> Load More Recommendations`;

                }

            }

        }
